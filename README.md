## Créer la table hotel avec une PK id
CREATE TABLE hotel (
	id INT NOT NULL AUTO_INCREMENT,
	nom VARCHAR(50) NOT NULL,
	nb_etoile INT NOT NULL,
	PRIMARY KEY ( id )
);
